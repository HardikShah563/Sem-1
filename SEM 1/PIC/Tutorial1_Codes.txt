#include<stdio.h>
#include<math.h>
void main()
{   
    float a, b, c, s, area;
    printf("Enter sides of a triangle: \n");
    printf("a: ");
    scanf("%f", &a);
    printf("b: ");
    scanf("%f", &b);
    printf("c: ");
    scanf("%f", &c);
    if( (a + b > c) && (b + c > a) && (c + a > b) )
    {
        s = (a + b + c)
        area = sqrt(  s * (s - a) * (s - b) * (s - c) );
        printf("The triangle can be formed, having area %.2f.", area);
    }
    else
    {
        printf("The triangle cannot be formed.");
    }
}

    for(i=1; i <= n1 && i <= n2; ++i)

        if(n1%i==0 && n2%i==0)
